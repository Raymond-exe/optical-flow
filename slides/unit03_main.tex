% SLIDE 4 - Four Single Integral Rules
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slideheader{Four Single Integral Quadrature Rules}{Theory}

\leftskip 0.7in \rightskip 0.7in
\vspace{0.1in}

\textbf{1. Trapezoidal Rule} \quad $O(h^2)$
\[
\int_a^b f\,dx \approx \frac{h}{2}\bigl[f_0 + 2\textstyle\sum f_i + f_n\bigr]
\]

\textbf{2. Simpson's 1/3 Rule} \quad $O(h^4)$, $n$ even
\[
\int_a^b f\,dx \approx \frac{h}{3}\bigl[f_0 + 4\textstyle\sum_{\text{odd}} f_i + 2\textstyle\sum_{\text{even}} f_i + f_n\bigr]
\]

\textbf{3. Simpson's 3/8 Rule} \quad $O(h^4)$, $n \bmod 3 = 0$
\[
\int_a^b f\,dx \approx \frac{3h}{8}\bigl[f_0 + 3\textstyle\sum_{i\not\equiv 0} f_i + 2\textstyle\sum_{i\equiv 0} f_i + f_n\bigr]
\]

\textbf{4. Midpoint Rule} \quad $O(h^2)$
\[
\int_a^b f\,dx \approx h \sum_{i=0}^{n-1} f\!\left(\frac{x_i + x_{i+1}}{2}\right)
\]

% SLIDE 5 - Double Integral + Differentiation
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slideheader{Double Integrals \& Numerical Differentiation}{Theory}

\leftskip 0.7in \rightskip 0.7in
\vspace{0.1in}

\begin{multicols}{2}
\textbf{Double Integration} (tensor product)


$\displaystyle
\hspace{-1.15in}\iint_R f(x,y)\,dx\,dy \approx \sum_i \sum_j w_{ij}\, f(x_i, y_j)
$

Two forms:
\begin{itemize}
  \item Trapezoidal 2D --- $O(h^2)$
  \item Simpson 2D --- $O(h^4)$
\end{itemize}

\columnbreak

\textbf{Finite Differences}

\[
\text{Central:} \qquad f'(x_i) \approx \frac{f_{i+1} - f_{i-1}}{2h}
\]

\[
\text{Forward:} \qquad f'(x_i) \approx \frac{f_{i+1} - f_i}{h}
\]

\[
\text{Backward:} \qquad f'(x_i) \approx \frac{f_i - f_{i-1}}{h}
\]
\end{multicols}

% SLIDE 7 - Lucas-Kanade Matrix Calculation
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slideheader{Matrix Calculation: Lucas-Kanade Optical Flow}{Optical Flow}

\leftskip 0.7in \rightskip 0.7in
\vspace{0.1in}

\textbf{Optical flow constraint:} \quad $I_x u + I_y v + I_t = 0$

\vspace{0.15in}
Over $N$ pixels in a window:
\[
\underbrace{\begin{bmatrix} I_{x_1} & I_{y_1} \\ I_{x_2} & I_{y_2} \\ \vdots & \vdots \\ I_{x_N} & I_{y_N} \end{bmatrix}}_{\mathbf{A}_{N \times 2}}
\begin{bmatrix} u \\ v \end{bmatrix}
= \underbrace{-\begin{bmatrix} I_{t_1} \\ I_{t_2} \\ \vdots \\ I_{t_N} \end{bmatrix}}_{\mathbf{b}_{N \times 1}}
\]

\vspace{0.15in}
\textbf{Least-squares solution:}
\[
\boxed{\begin{bmatrix} u \\ v \end{bmatrix} = \left(\mathbf{A}^T \mathbf{A}\right)^{-1} \mathbf{A}^T \mathbf{b}}
\]

$\mathbf{A}^T\mathbf{A}$ is a $2 \times 2$ matrix --- eigenvalues indicate flow reliability.

def integrate_trapezoidal(self, f, a, b, n):
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    y = np.array([f(xi) for xi in x])
    return h * (y[0]/2 + np.sum(y[1:-1]) + y[-1]/2)

def integrate_simpson13(self, f, a, b, n):
    if n % 2 != 0: n += 1
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    y = np.array([f(xi) for xi in x])
    return (h/3)*(y[0] + 4*np.sum(y[1::2])
           + 2*np.sum(y[2:-1:2]) + y[-1])
\end{lstlisting}

# Central difference
def differentiate(self, y, h, method='central'):
    dy = np.zeros(len(y))
    for i in range(1, len(y) - 1):
        dy[i] = (y[i+1] - y[i-1]) / (2*h)
    dy[0] = (y[1]-y[0])/h; dy[-1] = (y[-1]-y[-2])/h
    return dy

# Lucas-Kanade (in unit03.py)
A = np.column_stack([Ix, Iy])    # Nx2
ATA = A.T @ A                    # 2x2
ATb = A.T @ (-It)                # 2x1
flow = np.linalg.inv(ATA) @ ATb  # [u, v]

def optical_flow_lk(self, frame1, frame2, win=7):
    f1, f2 = np.float32(frame1), np.float32(frame2)
    f1 = cv2.GaussianBlur(f1, (5,5), 0)  # denoise
    f2 = cv2.GaussianBlur(f2, (5,5), 0)
    Ix = cv2.Sobel(f2, cv2.CV_32F, 1, 0, ksize=5)
    Iy = cv2.Sobel(f2, cv2.CV_32F, 0, 1, ksize=5)
    return self.lucas_kanade_matrix(Ix, Iy, f2-f1, win)

def lucas_kanade_matrix(self, Ix, Iy, It, win=7):
    # For each pixel window, solve 2x2 system directly:
    sxx, sxy, syy = dot(ix,ix), dot(ix,iy), dot(iy,iy)
    det = sxx*syy - sxy*sxy
    if det < tau*tau: continue  # skip singular
    u = (syy*bx - sxy*by) / det
    v = (sxx*by - sxy*bx) / det
\end{lstlisting}

\slideheader{Discussion and Conclusion}{}

\leftskip 0.7in \rightskip 0.7in
\vspace{0.05in}

\begin{multicols}{2}
\textbf{Key Findings}
\begin{itemize}
  \item Simpson's rules ($O(h^4)$) dramatically outperform Trapezoidal/Midpoint ($O(h^2)$) as $n$ increases
  \item Central differences give $O(h^2)$ derivative estimates --- sufficient for smooth data
  \item Differentiate-then-integrate pipeline introduces small cumulative drift
\end{itemize}

\columnbreak

\textbf{Conclusion}
\begin{itemize}
  \item Implemented 4 single + 2 double integral methods
  \item Applied to 2D optical flow (figure-8 trajectory)
  \item Lucas-Kanade matrix calc accurately recovers velocity via $(\mathbf{A}^T\mathbf{A})^{-1}\mathbf{A}^T\mathbf{b}$
  \item Complete motion-tracking pipeline: differentiate $\to$ flow solve $\to$ integrate
\end{itemize}
\end{multicols}
